# VuePress导航配置

在config文件下，修改nav,'/'表示当前项目docs所在目录

nav: [{
				text: 'Home',
				link: '/',
			},
			{
				text: 'Vuepress',
				link: '/page/vuepress/'
			},
			{
				text: 'Guide',
				link: '/guide/',
			},
			{
				text: 'Github',
				link: 'https://github.com/l328747807?tab=repositories'
			}
		],

# VuePress侧边栏配置

每个侧边栏与上面的nav对应，要修改多个侧边栏时，可参考如下设置：

sidebar: {
			'/guide/': [{
				title: '简介',
				collapsable: false,
				children: [
					'',
				]
			}],
			'/page/vuepress/': [{
					title: 'vuepress', // 一级菜单名称
					collapsable: true, // false为默认展开菜单, 默认值true是折叠,
					children: ['','vuepress01', 'vuepress02',
					]
				},
				{
					title: 'vuepress进阶',
					collapsable: true,
					children: [
						'',
						'test2.md'
					]
				}
			],
		}
